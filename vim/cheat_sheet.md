# Navigation
j,l,i,k = move left,up,down 

g = move to top of document
G = move to bottom of document
NG = move to Nth line

H = move to top of screen
L = move to bottom of screen
M = move to middle of screen

0 = move to start of line
$ = move to end of line

e = end of word
E = end of WORD
b = previous word
B = previous WORD
w = next word
W = next WORD

# Undo
u = undo
c-r = redo
